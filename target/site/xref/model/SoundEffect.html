1   package model;
2   import java.io.*;
3   import javax.sound.sampled.*;
4   public enum SoundEffect {
5   	/* The different resources... sounds generated with <a href="http://www.bfxr.net/">BFXR</a> */
6      COIN("/Sound/coin.wav"),
7      ENERGY("/Sound/energy.wav"),
8      SHOOT("/Sound/shoot.wav"),
9      GAMEOVER("/Sound/gameover.wav"),
10     BONK("/Sound/bonk.wav");
11      /* The current Audio clip */
12     private Clip clip;
13     
14      /**
15       * The initializer for a given sound effect
16       * @param soundFileName - the file to load
17       */
18     SoundEffect(String soundFileName) {
19        try {
20      	   /* get the right ressource from the JAR */
21      	 InputStream ressource = this.getClass().getResourceAsStream(soundFileName);
22      	   /* makes a inputstream from it*/
23           AudioInputStream audioInputStream = AudioSystem.getAudioInputStream(ressource);
24             /* get a channel, and load our sound effect inside */
25           clip = AudioSystem.getClip();
26           clip.open(audioInputStream);
27        } catch (UnsupportedAudioFileException e) {
28           e.printStackTrace();
29        } catch (IOException e) {
30           e.printStackTrace();
31        } catch (LineUnavailableException e) {
32           e.printStackTrace();
33        }
34     }
35     
36     /**
37      * This function starts the given sound effect
38      */
39     public void play() {
40  	       /* test if it is running */
41           if (clip.isRunning())
42              clip.stop();
43             /* reset the audio pos */
44           clip.setFramePosition(0);
45             /* start the effect */
46           clip.start();
47        	}
48     
49     /**
50      * This starts the sound system
51      */
52     static void init() {
53        values();
54     }
55  }
