8    /**
9     * This Daemon tries to move each step either horizontally or vertically, looking in which way the player is the futher away. It doesn't get as many times stucked as DaemonAI2, which in fact is his great brother.

12  public class DaemonAI1 implements DaemonAI {
13  	BufferedImage sprite;
14  	/**
15  	 * Initialise the AI
16  	 */
17  	public DaemonAI1(){
18  		try {
19  			this.sprite = ImageIO.read(this.getClass().getResourceAsStream("/Sprite/monster_1.png"));
20  			} 
21  		catch (IOException e) {
22  			e.printStackTrace();
23  			}
24  		}

26  	/**
27  	 * This function calculates the next horizontal position, depending on Lorann's position relative to that of the monster
28       * @param x - the X position of the monster
29       * @param y - the Y position of the monster
30       * @param lorannX - the X position of Lorann
31       * @param lorannY - the Y position of Lorann
32       * @return the next X position of the monster
33  	 */
34  	public int getNextX(int x, int y, int lorannX, int lorannY) {
35  		int deltaX=x-lorannX; /* The difference of X position. It's Abs(int) is the distance the monster is in horizontally, it's sign is the direction (left or right) the monster is gonna going in */
36  		int deltaY=y-lorannY; /* The difference of Y position. It's Abs(int) is the distance the monster is in vertically, it's sign is the direction (up or down) the monster is gonna going in */
37  		if(Math.abs(deltaX)>Math.abs(deltaY)){  //if the player is futher away horizontally than vertically
38  			if(deltaX<0){  // If the monster needs to go right 
39  				return x+1;
40  			    }
41  			if(deltaX>0){  // If the monster needs to go left
42  				return x-1;
43  			    }
44  		    }
45  		return x;	// Else, we don't move horizontally
46  		}
47  	
48  	/**
49  	 * This function calculates the next vertical position, depending on Lorann's position relative to that of the monster
50       * @param x - the X position of the monster
51       * @param y - the Y position of the monster
52       * @param lorannX - the X position of Lorann
53       * @param lorannY - the Y position of Lorann
54       * @return the next Y position of the monster
55  	 */
56  	public int getNextY(int x, int y, int lorannX, int lorannY) {
57  		int deltaX=x-lorannX; /* The difference of X position. It's Abs(int) is the distance the monster is in horizontally, it's sign is the direction (left or right) the monster is gonna going in */
58  		int deltaY=y-lorannY; /* The difference of Y position. It's Abs(int) is the distance the monster is in vertically, it's sign is the direction (up or down) the monster is gonna going in */
59  		if(Math.abs(deltaX)>Math.abs(deltaY)){  //if the player is futher away horizontally than vertically
60  			return y;	//We don't move vertically
61  		    }
62  		if(deltaY<0){	//Else, if the monster is left to the player...
63  			return y+1;  //... then move right
64  		    }
65  		if(deltaY>0){  //Else, if the monster is right to the player...
66  			return y-1;  //... then move left
67  		    }
68  		
69  		return y;  //Else, we are at the good vertical position, we don't move
70  		}
